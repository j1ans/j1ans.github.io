def allo(size):
	p.recvuntil("Command: ")
	p.sendline(str(1))

	p.recvuntil("Command: ")
	p.sendline(str(2))

	p.sendline(str(idx))

	p.sendline(str(size))
	p.recvuntil("Content: ")
	p.sendline(content)

def free(idx):
	p.recvuntil("Command: ")
	p.sendline(str(3))
	p.recvuntil("Index: ")
	p.sendline(str(idx))

def dump(idx):
	p.recvuntil("Command: ")
	p.sendline(str(4))
	p.recvuntil("Index: ")
	p.sendline(str(idx))
